## 从生活例子理解面向对象

**你写传奇脚本时，已经在用"对象"的思维了：**

```lua
player = {
    name = "战士",
    hp = 100,
    level = 30,
    attack = function(self)
        return self.level * 10
    end
}
```

这其实就是 Lua 中的"对象"——把数据（name、hp）和行为（attack）绑在一起。

C++ 的面向对象更正式、更强大，但核心思想是一样的。

---

## class 基础：把数据和函数绑在一起

**用 class 重写玩家：**

```cpp
#include <iostream>
#include <string>
using namespace std;

class Player {
public:                     // public 表示这些成员可以从外面访问
    string name;            // 数据成员
    int hp;
    int level;

    // 行为：攻击力计算
    int attack() {
        return level * 10;
    }

    // 行为：受到伤害
    void takeDamage(int dmg) {
        hp -= dmg;
        if (hp < 0) hp = 0;
    }

    // 行为：显示状态
    void show() {
        cout << name << " HP:" << hp
             << " 等级:" << level
             << " 攻击力:" << attack() << endl;
    }
};

int main() {
    Player p1;           // 创建对象（实例化）
    p1.name = "战士";
    p1.hp = 100;
    p1.level = 30;

    p1.show();
    p1.takeDamage(30);
    p1.show();

    return 0;
}
```

**输出：**
```
战士 HP:100 等级:30 攻击力:300
战士 HP:70 等级:30 攻击力:300
```

### class 和 struct 的区别

| 对比 | struct | class |
|------|--------|-------|
| 默认访问权限 | public | private（成员默认不可访问） |
| 传统用途 | 只存数据 | 数据+行为（面向对象） |
| C++ 里的实际区别 | 几乎一样，只是默认权限不同 | 同上 |

**经验法则：**
- 简单数据结构 → struct
- 有行为、有封装需求 → class

## 构造函数：创建对象时自动初始化

**问题：** 每次创建 Player 都要手动设置所有字段，很麻烦

**解决：** 用构造函数，对象创建时自动执行

```cpp
class Player {
public:
    string name;
    int hp;
    int level;

    // 构造函数：和类名相同，没有返回值
    Player(string n, int initHp, int initLevel) {
        name = n;
        hp = initHp;
        level = initLevel;
        cout << "玩家创建: " << name << endl;
    }

    void show() {
        cout << name << " HP:" << hp << " 等级:" << level << endl;
    }
};

int main() {
    Player p1("战士", 100, 30);   // 创建时就初始化
    Player p2("法师", 80, 25);

    p1.show();
    p2.show();

    return 0;
}
```

**输出：**
```
玩家创建: 战士
玩家创建: 法师
战士 HP:100 等级:30
法师 HP:80 等级:25
```

### 构造函数重载（多个构造函数）

```cpp
class Player {
public:
    string name;
    int hp;
    int level;

    // 构造函数1：完整参数
    Player(string n, int initHp, int initLevel) {
        name = n;
        hp = initHp;
        level = initLevel;
    }

    // 构造函数2：只给名字，默认等级1，HP100
    Player(string n) {
        name = n;
        hp = 100;
        level = 1;
    }

    // 构造函数3：无参构造函数
    Player() {
        name = "未命名";
        hp = 0;
        level = 0;
    }
};

int main() {
    Player p1("战士", 100, 30);  // 调用构造函数1
    Player p2("法师");            // 调用构造函数2
    Player p3;                   // 调用构造函数3

    return 0;
}
```

## 析构函数：对象销毁时自动调用

```cpp
class Player {
public:
    string name;

    // 构造函数
    Player(string n) {
        name = n;
        cout << name << " 加入了游戏" << endl;
    }

    // 析构函数：~ + 类名， 对象销毁时自动调用
    ~Player() {
        cout << name << " 离开了游戏" << endl;
    }
};

int main() {
    Player p1("战士");
    Player p2("法师");
    return 0;
}  // 函数结束，p1和p2被销毁，析构函数被调用
```

**输出：**
```
战士 加入了游戏
法师 加入了游戏
法师 离开了游戏
战士 离开了游戏
（后创建的后销毁，栈的顺序）
```

**什么时候用析构函数？**
- 释放动态申请的内存（`delete`）
- 关闭打开的文件
- 断开网络连接

## 继承：代码复用

**场景：** 战士、法师、道士都有"玩家"的通用属性

```cpp
// 基类（父类）：玩家共有的属性
class Player {
protected:    // protected：子类能访问，外部不能访问
    string name;
    int hp;
    int level;

public:
    Player(string n, int hpInit, int lvInit) {
        name = n;
        hp = hpInit;
        level = lvInit;
    }

    void takeDamage(int dmg) {
        hp -= dmg;
        if (hp < 0) hp = 0;
    }

    void show() {
        cout << name << " HP:" << hp << " 等级:" << level << endl;
    }
};

// 派生类（子类）：战士
class Warrior : public Player {
private:
    int armor;  // 战士特有的属性：护甲

public:
    Warrior(string n, int hpInit, int lvInit, int ar)
        : Player(n, hpInit, lvInit) {  // 调用父类构造函数
        armor = ar;
    }

    // 战士特有的技能：盾击
    void shieldBash() {
        cout << name << " 使用盾击！护甲抵消" << armor << "点伤害" << endl;
    }
};

int main() {
    Warrior w("战士", 100, 30, 20);
    w.show();        // 继承自 Player 的方法
    w.takeDamage(50); // 继承自 Player 的方法
    w.shieldBash();  // 战士自己的方法
    w.show();

    return 0;
}
```

**继承的层级关系：**
```
Player（基类）
├── Warrior（战士）：高护甲，近战
├── Mage（法师）：高魔攻，低血量
└── Priest（牧师）：治疗
```

### protected vs private

| 关键字 | 外部访问 | 子类访问 |
|--------|---------|---------|
| `private` | ❌ | ❌ |
| `protected` | ❌ | ✅ |
| `public` | ✅ | ✅ |

## 多态：同一接口，不同实现

**场景：** 战士攻击是物理伤害，法师攻击是魔法伤害

```cpp
class Player {
protected:
    string name;
    int attackPower;

public:
    Player(string n, int ap) : name(n), attackPower(ap) {}

    // 虚函数：子类可以重写
    virtual void attack() {
        cout << name << " 发起攻击！伤害:" << attackPower << endl;
    }

    virtual ~Player() {}  // 虚析构函数，删除子类对象时正确调用
};

class Warrior : public Player {
public:
    Warrior(string n) : Player(n, 50) {}  // 攻击力50

    void attack() override {  // override 明确表示重写
        cout << name << " 挥剑攻击！物理伤害:" << attackPower << endl;
    }
};

class Mage : public Player {
public:
    Mage(string n) : Player(n, 80) {}  // 攻击力80

    void attack() override {
        cout << name << " 释放火球术！魔法伤害:" << attackPower << endl;
    }
};

int main() {
    Player* players[3];
    players[0] = new Warrior("战士");
    players[1] = new Mage("法师");
    players[2] = new Player("未知玩家", 30);

    for (int i = 0; i < 3; i++) {
        players[i]->attack();  // 调用的是各自的 attack()
    }

    for (int i = 0; i < 3; i++) {
        delete players[i];
    }

    return 0;
}
```

**输出：**
```
战士 挥剑攻击！物理伤害:50
法师 释放火球术！魔法伤害:80
未知玩家 发起攻击！伤害:30
```

**关键点：**
- `virtual`：虚函数，子类可以覆盖
- `override`：明确表示要覆盖父类的方法
- 基类指针可以指向子类对象，调用时自动调用正确版本

## 纯虚函数和抽象类

**纯虚函数：** 只有声明，没有实现，强制子类必须重写

```cpp
class Character {
public:
    string name;

    // 纯虚函数：= 0 表示没有默认实现
    virtual void attack() = 0;
    virtual ~Character() {}
};

// 不能直接创建 Character 对象，它是"抽象"的
// Character c;  // 错误！

class Knight : public Character {
public:
    Knight(string n) { name = n; }

    void attack() override {
        cout << name << " 挥剑斩击！" << endl;
    }
};
```

**抽象类的用途：** 定义"接口"，规定子类必须有哪些方法。

## 练习：角色系统

**题目：** 用类继承实现以下结构：
- 基础角色有 name, hp, attack()
- 战士：高血量，近战攻击
- 弓箭手：中血量，远程攻击
- 法师：低血量，高攻击

```cpp
#include <iostream>
#include <string>
using namespace std;

class Character {
protected:
    string name;
    int hp;

public:
    Character(string n, int h) : name(n), hp(h) {}
    virtual void attack() = 0;  // 纯虚
    virtual void show() {
        cout << name << " HP:" << hp;
    }
    virtual ~Character() {}
};

class Warrior : public Character {
public:
    Warrior(string n) : Character(n, 150) {}

    void attack() override {
        cout << name << " 挥剑斩击！" << endl;
    }
};

class Archer : public Character {
public:
    Archer(string n) : Character(n, 100) {}

    void attack() override {
        cout << name << " 拉弓射箭！" << endl;
    }
};

class Mage : public Character {
public:
    Mage(string n) : Character(n, 80) {}

    void attack() override {
        cout << name << " 施放奥术飞弹！" << endl;
    }
};

int main() {
    Character* team[3];
    team[0] = new Warrior("战士");
    team[1] = new Archer("弓箭手");
    team[2] = new Mage("法师");

    for (int i = 0; i < 3; i++) {
        team[i]->show();
        team[i]->attack();
        cout << endl;
    }

    for (int i = 0; i < 3; i++) delete team[i];

    return 0;
}
```

## 本周总结

| 概念 | 记住一句话 |
|------|-----------|
| class | 把数据和函数封装在一起 |
| public / private / protected | 访问权限控制 |
| 构造函数 | 创建对象时自动调用的初始化函数 |
| 析构函数 | 对象销毁时自动调用（清理资源） |
| 继承 | 子类复用父类的代码 |
| 虚函数 virtual | 子类可以覆盖父类的实现 |
| 多态 | 同一接口（基类指针）调用不同实现 |
| 纯虚函数 = 0 | 强制子类实现，定义接口 |
